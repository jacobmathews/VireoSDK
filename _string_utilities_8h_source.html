337 class CompareSubString
338 {
339  public:
340  Boolean operator()(const SubString &a, const SubString &b) const {
341  Int32 aSize = a.Length();
342  Int32 bSize = b.Length();
343  if (aSize != bSize) {
344  return aSize < bSize;
345  } else {
346  const Utf8Char* ba = a.Begin();
347  const Utf8Char* bb = b.Begin();
348  const Utf8Char* ea = a.End();
349 
350  while (ba < ea) {
351  if (*ba != *bb) {
352  return *ba < *bb;
353  } else {
354  ba++;
355  bb++;
356  }
357  }
358  return false;
359  }
360  }
